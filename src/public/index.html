Developer Experience

Clean, Simple API

Get rental data with a single POST request.

POST https://rentvolt-api.onrender.com/api/scrape/listings
Content-Type: application/json
x-api-key: YOUR_API_KEY

{
  "city": "oakland",
  "state": "ca",
  "filters": {
    "maxPrice": 3000,
    "minBeds": 2,
    "sortBy": "price",
    "limit": 20
  }
}
{
  "success": true,
  "listings": [
    {
      "source": "apartments.com",
      "price": 1850,
      "address": "420 Grand Ave, Oakland, CA",
      "beds": "2 bed",
      "baths": "1 bath"
    },
    // ... more listings
  ],
  "total": 20,
  "sources": { "apartments": 8, "zillow": 5, ... }
}
